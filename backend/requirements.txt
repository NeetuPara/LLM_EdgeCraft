#   1. pip install unsloth
#      (installs transformers, trl, peft, datasets -- but pulls CPU-only torch from PyPI)
#   2. pip install torch==2.7.0+cu128 torchvision torchaudio \
#         --index-url https://download.pytorch.org/whl/cu128 \
#         --force-reinstall --no-deps
#      (restores CUDA torch that unsloth replaced -- CRITICAL step)
#   3. pip install -r requirements.txt
#
# WHY: unsloth's dependency resolver pulls torch from PyPI (CPU-only).
#      Step 2 force-reinstalls the CUDA build without re-triggering that resolver.
#
# Use .\scripts\install.ps1 (Windows) or ./scripts/install.sh (Linux/Mac)
# which handles this automatically.
